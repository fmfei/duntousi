def drop_breadcrumb(title=nil, url=nil)
  if title == 'backend/home'
    title = nil
    return
  end
  title ||= t(@page_title)
  url ||= url_for
  if title
    begin
      if url
        @breadcrumbs.push("<a href=\"#{url}\">#{t(title)}</a>".html_safe)
      else
        @breadcrumbs.push("<h1>#{t(title)}</h1>".html_safe)
      end
      
    rescue Exception => e
      
    end
  end
end
