def self.save(rules)
  result = true
  begin
    File.open("#{Rails.root}/config/front_config.yml", 'w') { |f|
      #YAML.dump(rules, f)
      f.puts rules.ya2yaml
    }
 rescue => err
   logger = Logger.new("#{Rails.root}/log/err.log")
   logger.error err
   logger.close     
   result = false
 end
 result
end
